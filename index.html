Quick Installation

v1.2.x

Integrate TechCard in seconds. This new version supports **TypeScript** out of the box and is served via a global CDN. 

<!-- Import TechCard -->
<script src="https://cdn.jsdelivr.net/gh/Tostenn/TechCard/dist/techcard.js"></script>

<script>
  window.onload = () => {
    const myCard = new window.TechCard({
      card: "solo",
      user: {
        name: "Your Name",
        role: "Full-Stack Developer",
        avatar: "https://your-image.jpg"
      }
    });
    myCard.open();
  }
</script>
