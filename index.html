Installation

# Modern (npm)

npm install techcard

# Simple (CDN)

techcard@1.0.2/dist/techcard.js 
<!-- Usage Example -->
<script src="https://cdn.jsdelivr.net/npm/techcard@1.0.2/dist/techcard.js"></script>

<script>
  window.onload = () => {
    const myCard = new window.TechCard({
      card: "solo",
      theme: "dark",
      user: {
        name: "Tosten Kouya",
        role: "Full-Stack Developer"
      }
    });
    myCard.open();
  }
</script>
