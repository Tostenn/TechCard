Quick Installation

Integrate TechCard in seconds. Copy the snippet below and paste it before the </body> tag of your site. 

<!-- Import TechCard via CDN -->
<script src="https://cdn.jsdelivr.net/gh/Tostenn/TechCard/build/techcard.js"></script>

<script>
  window.onload = () => {
    new window.TechCard({
      card: "solo",
      user: {
        name: "Your Name",
        role: "Full-Stack Developer",
        avatar: "https://your-image.jpg"
      }
    }).open();
  }
</script>
